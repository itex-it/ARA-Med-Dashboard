        event_type: z.string().min(1, 'Dieses Feld ist erforderlich.'),

        channel_target: z.string().optional(),
        fallback_channel: z.enum(['inbox','email','telegram','sms']).optional(),
        fallback_channel_target: z.string().optional(),
        template_id: z.string().uuid().optional().nullable(),
        priority: z.enum(['high','normal','low']).default('normal'),
        time_window_from: z.string().regex(/^\d{2}:\d{2}$/).optional().nullable(),
        time_window_until: z.string().regex(/^\d{2}:\d{2}$/).optional().nullable(),
        retry_interval_minutes: z.number().int().positive().optional().nullable(),
        max_retries: z.number().int().min(1).max(10).default(3),
        privacy_class: z.enum(['standard','restricted','minimal']).default('standard'),
      }).refine(d => {
        if (d.direction === 'patient' && d.channel === 'inbox') return false
        if (d.direction === 'patient' && d.channel === 'telegram') return false
        return true
      }, { message: 'Patientenbenachrichtigungen unterstützen nur E-Mail und SMS.', path: ['channel'] })
    - createCommRuleSchema: commRuleBaseSchema
    - updateCommRuleSchema: commRuleBaseSchema.extend({ id: z.string().uuid() })
    - deleteCommRuleSchema: z.object({ id: z.string().uuid() })
    - toggleCommRuleSchema: z.object({ id: z.string().uuid(), active: z.boolean() })

    toggleCommRuleAction: update { active, updated_at } with both eq filters; return { success: true, active: parsed.data.active }.

    All error paths: console.error('[actionName] DB error:', dbError)
