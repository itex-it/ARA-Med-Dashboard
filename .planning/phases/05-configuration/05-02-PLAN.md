    addDeputyPeriodAction(data: { start_date: string; end_date: string; label?: string }): Auth + role gate + Zod refinement validation. Insert into deputy_periods. Return state.

    deleteDeputyPeriodAction(data: { id: string }): Auth + role gate. Delete with both id and tenant_id filters.
