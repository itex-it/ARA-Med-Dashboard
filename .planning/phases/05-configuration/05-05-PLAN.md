    Create 'src/app/actions/deputy.ts' with 'use server' directive.

    const ALLOWED_ROLES = ['operator', 'ordination_admin']

    Action state interfaces:
    - export interface DeputyDoctorActionState { success?: boolean; error?: string }

    Shared field schema (used by both add and update):
    deputyFieldsSchema with all fields:
    - name: z.string().min(1, 'Dieses Feld ist erforderlich.')
    - forwarding_number: z.string().min(1, 'Dieses Feld ist erforderlich.')
    - greeting_text: z.string().optional()
    - start_date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/).nullable().optional()
    - end_date: z.string().regex(/^\d{4}-\d{2}-\d{2}$/).nullable().optional()
    - pid_zero_behavior: z.enum(['refuse', 'waitlist', 'book_normal', 'forward'])
    - pid_zero_forward_number: z.string().optional().nullable()
    - own_service_active: z.boolean().default(false)
    - own_service_prompt: z.string().optional()
    - own_service_pid_zero_behavior: z.enum(['refuse','waitlist','book_normal','forward']).optional().nullable()

    addDeputyDoctorSchema = deputyFieldsSchema.refine(
      d => !d.start_date || !d.end_date || d.end_date >= d.start_date,
      { message: 'Das Enddatum muss nach dem Startdatum liegen.', path: ['end_date'] }
    )

    updateDeputyDoctorSchema = z.object({ id: z.string().uuid() }).merge(deputyFieldsSchema.partial()).refine(same date range logic)

    deleteDeputyDoctorSchema = z.object({ id: z.string().uuid() })

    addDeputyDoctorAction(data: unknown): Full 5-step auth sequence. Zod validate with addDeputyDoctorSchema. serviceClient.from('deputy_doctors').insert({ tenant_id: tenantId, ...parsed.data }). Return { success: true } or error.

    updateDeputyDoctorAction(data: unknown): Full auth sequence. Zod validate with updateDeputyDoctorSchema. Build updateData from parsed.data excluding 'id' key. Ensure updated_at: new Date().toISOString() in updateData. serviceClient.from('deputy_doctors').update(updateData).eq('id', parsed.data.id).eq('tenant_id', tenantId). Both eq filters mandatory.

    deleteDeputyDoctorAction(data: unknown): Full auth sequence. Zod validate with deleteDeputyDoctorSchema. serviceClient.from('deputy_doctors').delete().eq('id', parsed.data.id).eq('tenant_id', tenantId). Both filters mandatory.

    All error paths: console.error('[functionName] DB error:', dbError).
  </action>
